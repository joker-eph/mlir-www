31 void OpMethodSignature::writeDefTo(raw_ostream &os,
32  StringRef namePrefix) const {
33  // We need to remove the default values for parameters in method definition.
34  // TODO: We are using '=' and ',' as delimiters for parameter
35  // initializers. This is incorrect for initializer list with more than one
36  // element. Change to a more robust approach.
37  auto removeParamDefaultValue = [](StringRef params) {
38  std::string result;
39  std::pair<StringRef, StringRef> parts;
40  while (!params.empty()) {
41  parts = params.split("=");
42  result.append(result.empty() ? "" : ", ");
43  result += parts.first;
44  params = parts.second.split(",").second;
45  }
46  return result;
47  };
48 
49  os << returnType << (elideSpaceAfterType(returnType) ? "" : " ") << namePrefix
50  << (namePrefix.empty() ? "" : "::") << methodName << "("
51  << removeParamDefaultValue(parameters) << ")";
52 }
