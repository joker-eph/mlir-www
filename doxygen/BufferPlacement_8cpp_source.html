9 // This file implements logic for computing correct alloc and dealloc positions.
10 // Furthermore, buffer placement also adds required new alloc and copy
11 // operations to ensure that all buffers are deallocated.The main class is the
12 // BufferPlacementPass class that implements the underlying algorithm. In order
13 // to put allocations and deallocations at safe positions, it is significantly
14 // important to put them into the correct blocks. However, the liveness analysis
15 // does not pay attention to aliases, which can occur due to branches (and their
16 // associated block arguments) in general. For this purpose, BufferPlacement
17 // firstly finds all possible aliases for a single value (using the
18 // BufferPlacementAliasAnalysis class). Consider the following example:
19 //
20 // ^bb0(%arg0):
21 // cond_br %cond, ^bb1, ^bb2
22 // ^bb1:
23 // br ^exit(%arg0)
24 // ^bb2:
25 // %new_value = ...
26 // br ^exit(%new_value)
27 // ^exit(%arg1):
28 // return %arg1;
29 //
30 // Using liveness information on its own would cause us to place the allocs and
31 // deallocs in the wrong block. This is due to the fact that %new_value will not
32 // be liveOut of its block. Instead, we can place the alloc for %new_value
33 // in bb0 and its associated dealloc in exit. Alternatively, the alloc can stay
34 // (or even has to stay due to additional dependencies) at this location and we
35 // have to free the buffer in the same block, because it cannot be freed in the
36 // post dominator. However, this requires a new copy buffer for %arg1 that will
37 // contain the actual contents. Using the class BufferPlacementAliasAnalysis, we
38 // will find out that %new_value has a potential alias %arg1. In order to find
39 // the dealloc position we have to find all potential aliases, iterate over
40 // their uses and find the common post-dominator block (note that additional
41 // copies and buffers remove potential aliases and will influence the placement
42 // of the deallocs). In all cases, the computed block can be safely used to free
43 // the %new_value buffer (may be exit or bb2) as it will die and we can use
44 // liveness information to determine the exact operation after which we have to
45 // insert the dealloc. Finding the alloc position is similar and non-obvious.
46 // However, the algorithm supports moving allocs to other places and introducing
47 // copy buffers and placing deallocs in safe places to ensure that all buffers
48 // will be freed in the end.
49 //
50 // TODO:
51 // The current implementation does not support loops and the resulting code will
52 // be invalid with respect to program semantics. The only thing that is
53 // currently missing is a high-level loop analysis that allows us to move allocs
54 // and deallocs outside of the loop blocks. Furthermore, it doesn't also accept
55 // functions which return buffers already.
56 //

69 /// Walks over all immediate return-like terminators in the given region.
70 template <typename FuncT>
71 static void walkReturnOperations(Region *region, const FuncT &func) {
72  for (Block &block : *region)
73  for (Operation &operation : block) {
74  // Skip non-return-like terminators.
75  if (operation.hasTrait<OpTrait::ReturnLike>())
76  func(&operation);
77  }
78 }

85 /// A straight-forward alias analysis which ensures that all aliases of all
86 /// values will be determined. This is a requirement for the BufferPlacement
87 /// class since you need to determine safe positions to place alloc and
88 /// deallocs.
89 class BufferPlacementAliasAnalysis {
90 public:
91  using ValueSetT = SmallPtrSet<Value, 16>;
92  using ValueMapT = llvm::DenseMap<Value, ValueSetT>;
93 
94 public:
95  /// Constructs a new alias analysis using the op provided.
96  BufferPlacementAliasAnalysis(Operation *op) { build(op); }
97 
98  /// Find all immediate aliases this value could potentially have.
99  ValueMapT::const_iterator find(Value value) const {
100  return aliases.find(value);
101  }
102 
103  /// Returns the end iterator that can be used in combination with find.
104  ValueMapT::const_iterator end() const { return aliases.end(); }
105 
106  /// Find all immediate and indirect aliases this value could potentially
107  /// have. Note that the resulting set will also contain the value provided as
108  /// it is an alias of itself.
109  ValueSetT resolve(Value value) const {
110  ValueSetT result;
111  resolveRecursive(value, result);
112  return result;
113  }
114 
115  /// Removes the given values from all alias sets.
116  void remove(const SmallPtrSetImpl<Value> &aliasValues) {
117  for (auto &entry : aliases)
118  llvm::set_subtract(entry.second, aliasValues);
119  }
120 
121 private:
122  /// Recursively determines alias information for the given value. It stores
123  /// all newly found potential aliases in the given result set.
124  void resolveRecursive(Value value, ValueSetT &result) const {
125  if (!result.insert(value).second)
126  return;
127  auto it = aliases.find(value);
128  if (it == aliases.end())
129  return;
130  for (Value alias : it->second)
131  resolveRecursive(alias, result);
132  }
133 
134  /// This function constructs a mapping from values to its immediate aliases.
135  /// It iterates over all blocks, gets their predecessors, determines the
136  /// values that will be passed to the corresponding block arguments and
137  /// inserts them into the underlying map. Furthermore, it wires successor
138  /// regions and branch-like return operations from nested regions.
139  void build(Operation *op) {
140  // Registers all aliases of the given values.
141  auto registerAliases = [&](auto values, auto aliases) {
142  for (auto entry : llvm::zip(values, aliases))
143  this->aliases[std::get<0>(entry)].insert(std::get<1>(entry));
144  };
145 
146  // Add additional aliases created by view changes to the alias list.
147  op->walk([&](ViewLikeOpInterface viewInterface) {
148  aliases[viewInterface.getViewSource()].insert(
149  viewInterface.getOperation()->getResult(0));
150  });
151 
152  // Query all branch interfaces to link block argument aliases.
153  op->walk([&](BranchOpInterface branchInterface) {
154  Block *parentBlock = branchInterface.getOperation()->getBlock();
155  for (auto it = parentBlock->succ_begin(), e = parentBlock->succ_end();
156  it != e; ++it) {
157  // Query the branch op interface to get the successor operands.
158  auto successorOperands =
159  branchInterface.getSuccessorOperands(it.getIndex());
160  if (!successorOperands.hasValue())
161  continue;
162  // Build the actual mapping of values to their immediate aliases.
163  registerAliases(successorOperands.getValue(), (*it)->getArguments());
164  }
165  });
166 
167  // Query the RegionBranchOpInterface to find potential successor regions.
168  op->walk([&](RegionBranchOpInterface regionInterface) {
169  // Create an empty attribute for each operand to comply with the
170  // `getSuccessorRegions` interface definition that requires a single
171  // attribute per operand.
172  SmallVector<Attribute, 2> operandAttributes(
173  regionInterface.getOperation()->getNumOperands());
174 
175  // Extract all entry regions and wire all initial entry successor inputs.
176  SmallVector<RegionSuccessor, 2> entrySuccessors;
177  regionInterface.getSuccessorRegions(/*index=*/llvm::None,
178  operandAttributes, entrySuccessors);
179  for (RegionSuccessor &entrySuccessor : entrySuccessors) {
180  // Wire the entry region's successor arguments with the initial
181  // successor inputs.
182  assert(entrySuccessor.getSuccessor() &&
183  "Invalid entry region without an attached successor region");
184  registerAliases(regionInterface.getSuccessorEntryOperands(
185  entrySuccessor.getSuccessor()->getRegionNumber()),
186  entrySuccessor.getSuccessorInputs());
187  }
188 
189  // Wire flow between regions and from region exits.
190  for (Region &region : regionInterface.getOperation()->getRegions()) {
191  // Iterate over all successor region entries that are reachable from the
192  // current region.
193  SmallVector<RegionSuccessor, 2> successorRegions;
194  regionInterface.getSuccessorRegions(
195  region.getRegionNumber(), operandAttributes, successorRegions);
196  for (RegionSuccessor &successorRegion : successorRegions) {
197  // Iterate over all immediate terminator operations and wire the
198  // successor inputs with the operands of each terminator.
199  walkReturnOperations(&region, [&](Operation *terminator) {
200  registerAliases(terminator->getOperands(),
201  successorRegion.getSuccessorInputs());
202  });
203  }
204  }
205  });
206  }
207 
208  /// Maps values to all immediate aliases this value can have.
209  ValueMapT aliases;
210 };

216 // The main buffer placement analysis used to place allocs, copies and deallocs.
217 class BufferPlacement {
218 public:
219  using ValueSetT = BufferPlacementAliasAnalysis::ValueSetT;

221  /// An intermediate representation of a single allocation node.
222  struct AllocEntry {
223  /// A reference to the associated allocation node.
224  Value allocValue;
225 
226  /// The associated placement block in which the allocation should be
227  /// performed.
228  Block *placementBlock;
229 
230  /// The associated dealloc operation (if any).
231  Operation *deallocOperation;
232  };
233 
234  using AllocEntryList = SmallVector<AllocEntry, 8>;
235 
236 public:
237  BufferPlacement(Operation *op)
238  : operation(op), aliases(op), liveness(op), dominators(op),
239  postDominators(op) {
240  // Gather all allocation nodes
241  initBlockMapping();
242  }
243 
244  /// Performs the actual placement/creation of all alloc, copy and dealloc
245  /// nodes.
246  void place() {
247  // Place all allocations.
248  placeAllocs();
249  // Add additional allocations and copies that are required.
250  introduceCopies();
251  // Find all associated dealloc nodes.
252  findDeallocs();
253  // Place deallocations for all allocation entries.
254  placeDeallocs();
255  }

257 private:
258  /// Initializes the internal block mapping by discovering allocation nodes. It
259  /// maps all allocation nodes to their initial block in which they can be
260  /// safely allocated.
261  void initBlockMapping() {
262  operation->walk([&](MemoryEffectOpInterface opInterface) {
263  // Try to find a single allocation result.
264  SmallVector<MemoryEffects::EffectInstance, 2> effects;
265  opInterface.getEffects(effects);
266 
267  SmallVector<MemoryEffects::EffectInstance, 2> allocateResultEffects;
268  llvm::copy_if(
269  effects, std::back_inserter(allocateResultEffects),
270  [=](MemoryEffects::EffectInstance &it) {
271  Value value = it.getValue();
272  return isa<MemoryEffects::Allocate>(it.getEffect()) && value &&
273  value.isa<OpResult>() &&
274  it.getResource() !=
275  SideEffects::AutomaticAllocationScopeResource::get();
276  });
277  // If there is one result only, we will be able to move the allocation and
278  // (possibly existing) deallocation ops.
279  if (allocateResultEffects.size() != 1)
280  return;
281  // Get allocation result.
282  auto allocResult = allocateResultEffects[0].getValue().cast<OpResult>();
283  // Find the initial allocation block and register this result.
284  allocs.push_back(
285  {allocResult, getInitialAllocBlock(allocResult), nullptr});
286  });
287  }
288 
289  /// Computes a valid allocation position in a dominator (if possible) for the
290  /// given allocation result.
291  Block *getInitialAllocBlock(OpResult result) {
292  // Get all allocation operands as these operands are important for the
293  // allocation operation.
294  Operation *owner = result.getOwner();
295  auto operands = owner->getOperands();
296  Block *dominator;
297  if (operands.size() < 1)
298  dominator =
299  findCommonDominator(result, aliases.resolve(result), dominators);
300  else {
301  // If this node has dependencies, check all dependent nodes with respect
302  // to a common post dominator in which all values are available.
303  ValueSetT dependencies(++operands.begin(), operands.end());
304  dominator =
305  findCommonDominator(*operands.begin(), dependencies, postDominators);
306  }
307 
308  // Do not move allocs out of their parent regions to keep them local.
309  if (dominator->getParent() != owner->getParentRegion())
310  return &owner->getParentRegion()->front();
311  return dominator;
312  }

314  /// Finds correct alloc positions according to the algorithm described at
315  /// the top of the file for all alloc nodes that can be handled by this
316  /// analysis.
317  void placeAllocs() const {
318  for (const AllocEntry &entry : allocs) {
319  Value alloc = entry.allocValue;
320  // Get the actual block to place the alloc and get liveness information
321  // for the placement block.
322  Block *placementBlock = entry.placementBlock;
323  // We have to ensure that we place the alloc before its first use in this
324  // block.
325  const LivenessBlockInfo *livenessInfo =
326  liveness.getLiveness(placementBlock);
327  Operation *startOperation = livenessInfo->getStartOperation(alloc);
328  // Check whether the start operation lies in the desired placement block.
329  // If not, we will use the terminator as this is the last operation in
330  // this block.
331  if (startOperation->getBlock() != placementBlock)
332  startOperation = placementBlock->getTerminator();
333 
334  // Move the alloc in front of the start operation.
335  Operation *allocOperation = alloc.getDefiningOp();
336  allocOperation->moveBefore(startOperation);
337  }
338  }
339 
340  /// Introduces required allocs and copy operations to avoid memory leaks.
341  void introduceCopies() {
342  // Initialize the set of values that require a dedicated memory free
343  // operation since their operands cannot be safely deallocated in a post
344  // dominator.
345  SmallPtrSet<Value, 8> valuesToFree;
346  llvm::SmallDenseSet<std::tuple<Value, Block *>> visitedValues;
347  SmallVector<std::tuple<Value, Block *>, 8> toProcess;
348 
349  // Check dominance relation for proper dominance properties. If the given
350  // value node does not dominate an alias, we will have to create a copy in
351  // order to free all buffers that can potentially leak into a post
352  // dominator.
353  auto findUnsafeValues = [&](Value source, Block *definingBlock) {
354  auto it = aliases.find(source);
355  if (it == aliases.end())
356  return;
357  for (Value value : it->second) {
358  if (valuesToFree.count(value) > 0)
359  continue;
360  // Check whether we have to free this particular block argument.
361  if (!dominators.dominates(definingBlock, value.getParentBlock())) {
362  toProcess.emplace_back(value, value.getParentBlock());
363  valuesToFree.insert(value);
364  } else if (visitedValues.insert(std::make_tuple(value, definingBlock))
365  .second)
366  toProcess.emplace_back(value, definingBlock);
367  }
368  };
369 
370  // Detect possibly unsafe aliases starting from all allocations.
371  for (auto &entry : allocs)
372  findUnsafeValues(entry.allocValue, entry.placementBlock);
373 
374  // Try to find block arguments that require an explicit free operation
375  // until we reach a fix point.
376  while (!toProcess.empty()) {
377  auto current = toProcess.pop_back_val();
378  findUnsafeValues(std::get<0>(current), std::get<1>(current));
379  }
380 
381  // Update buffer aliases to ensure that we free all buffers and block
382  // arguments at the correct locations.
383  aliases.remove(valuesToFree);
384 
385  // Add new allocs and additional copy operations.
386  for (Value value : valuesToFree) {
387  if (auto blockArg = value.dyn_cast<BlockArgument>())
388  introduceBlockArgCopy(blockArg);
389  else
390  introduceValueCopyForRegionResult(value);
391 
392  // Register the value to require a final dealloc. Note that we do not have
393  // to assign a block here since we do not want to move the allocation node
394  // to another location.
395  allocs.push_back({value, nullptr, nullptr});
396  }
397  }

399  /// Introduces temporary allocs in all predecessors and copies the source
400  /// values into the newly allocated buffers.
401  void introduceBlockArgCopy(BlockArgument blockArg) {
402  // Allocate a buffer for the current block argument in the block of
403  // the associated value (which will be a predecessor block by
404  // definition).
405  Block *block = blockArg.getOwner();
406  for (auto it = block->pred_begin(), e = block->pred_end(); it != e; ++it) {
407  // Get the terminator and the value that will be passed to our
408  // argument.
409  Operation *terminator = (*it)->getTerminator();
410  auto branchInterface = cast<BranchOpInterface>(terminator);
411  // Query the associated source value.
412  Value sourceValue =
413  branchInterface.getSuccessorOperands(it.getSuccessorIndex())
414  .getValue()[blockArg.getArgNumber()];
415  // Create a new alloc and copy at the current location of the terminator.
416  Value alloc = introduceBufferCopy(sourceValue, terminator);
417  // Wire new alloc and successor operand.
418  auto mutableOperands =
419  branchInterface.getMutableSuccessorOperands(it.getSuccessorIndex());
420  if (!mutableOperands.hasValue())
421  terminator->emitError() << "terminators with immutable successor "
422  "operands are not supported";
423  else
424  mutableOperands.getValue()
425  .slice(blockArg.getArgNumber(), 1)
426  .assign(alloc);
427  }
428 
429  // Check whether the block argument has implicitly defined predecessors via
430  // the RegionBranchOpInterface. This can be the case if the current block
431  // argument belongs to the first block in a region and the parent operation
432  // implements the RegionBranchOpInterface.
433  Region *argRegion = block->getParent();
434  RegionBranchOpInterface regionInterface;
435  if (!argRegion || &argRegion->front() != block ||
436  !(regionInterface =
437  dyn_cast<RegionBranchOpInterface>(argRegion->getParentOp())))
438  return;
439 
440  introduceCopiesForRegionSuccessors(
441  regionInterface, argRegion->getParentOp()->getRegions(),
442  [&](RegionSuccessor &successorRegion) {
443  // Find a predecessor of our argRegion.
444  return successorRegion.getSuccessor() == argRegion;
445  },
446  [&](RegionSuccessor &successorRegion) {
447  // The operand index will be the argument number.
448  return blockArg.getArgNumber();
449  });
450  }

452  /// Introduces temporary allocs in front of all associated nested-region
453  /// terminators and copies the source values into the newly allocated buffers.
454  void introduceValueCopyForRegionResult(Value value) {
455  // Get the actual result index in the scope of the parent terminator.
456  Operation *operation = value.getDefiningOp();
457  auto regionInterface = cast<RegionBranchOpInterface>(operation);
458  introduceCopiesForRegionSuccessors(
459  regionInterface, operation->getRegions(),
460  [&](RegionSuccessor &successorRegion) {
461  // Determine whether this region has a successor entry that leaves
462  // this region by returning to its parent operation.
463  return !successorRegion.getSuccessor();
464  },
465  [&](RegionSuccessor &successorRegion) {
466  // Find the associated success input index.
467  return llvm::find(successorRegion.getSuccessorInputs(), value)
468  .getIndex();
469  });
470  }

472  /// Introduces buffer copies for all terminators in the given regions. The
473  /// regionPredicate is applied to every successor region in order to restrict
474  /// the copies to specific regions. Thereby, the operandProvider is invoked
475  /// for each matching region successor and determines the operand index that
476  /// requires a buffer copy.
477  template <typename TPredicate, typename TOperandProvider>
478  void
479  introduceCopiesForRegionSuccessors(RegionBranchOpInterface regionInterface,
480  MutableArrayRef<Region> regions,
481  const TPredicate &regionPredicate,
482  const TOperandProvider &operandProvider) {
483  // Create an empty attribute for each operand to comply with the
484  // `getSuccessorRegions` interface definition that requires a single
485  // attribute per operand.
486  SmallVector<Attribute, 2> operandAttributes(
487  regionInterface.getOperation()->getNumOperands());
488  for (Region &region : regions) {
489  // Query the regionInterface to get all successor regions of the current
490  // one.
491  SmallVector<RegionSuccessor, 2> successorRegions;
492  regionInterface.getSuccessorRegions(region.getRegionNumber(),
493  operandAttributes, successorRegions);
494  // Try to find a matching region successor.
495  RegionSuccessor *regionSuccessor =
496  llvm::find_if(successorRegions, regionPredicate);
497  if (regionSuccessor == successorRegions.end())
498  continue;
499  // Get the operand index in the context of the current successor input
500  // bindings.
501  auto operandIndex = operandProvider(*regionSuccessor);
502 
503  // Iterate over all immediate terminator operations to introduce
504  // new buffer allocations. Thereby, the appropriate terminator operand
505  // will be adjusted to point to the newly allocated buffer instead.
506  walkReturnOperations(&region, [&](Operation *terminator) {
507  // Extract the source value from the current terminator.
508  Value sourceValue = terminator->getOperand(operandIndex);
509  // Create a new alloc at the current location of the terminator.
510  Value alloc = introduceBufferCopy(sourceValue, terminator);
511  // Wire alloc and terminator operand.
512  terminator->setOperand(operandIndex, alloc);
513  });
514  }
515  }
516 
517  /// Creates a new memory allocation for the given source value and copies
518  /// its content into the newly allocated buffer. The terminator operation is
519  /// used to insert the alloc and copy operations at the right places.
520  Value introduceBufferCopy(Value sourceValue, Operation *terminator) {
521  // Create a new alloc at the current location of the terminator.
522  auto memRefType = sourceValue.getType().cast<MemRefType>();
523  OpBuilder builder(terminator);
524 
525  // Extract information about dynamically shaped types by
526  // extracting their dynamic dimensions.
527  SmallVector<Value, 4> dynamicOperands;
528  for (auto shapeElement : llvm::enumerate(memRefType.getShape())) {
529  if (!ShapedType::isDynamic(shapeElement.value()))
530  continue;
531  dynamicOperands.push_back(builder.create<DimOp>(
532  terminator->getLoc(), sourceValue, shapeElement.index()));
533  }
534 
535  // TODO: provide a generic interface to create dialect-specific
536  // Alloc and CopyOp nodes.
537  auto alloc = builder.create<AllocOp>(terminator->getLoc(), memRefType,
538  dynamicOperands);
539 
540  // Create a new copy operation that copies to contents of the old
541  // allocation to the new one.
542  builder.create<linalg::CopyOp>(terminator->getLoc(), sourceValue, alloc);
543 
544  return alloc;
545  }

547  /// Finds associated deallocs that can be linked to our allocation nodes (if
548  /// any).
549  void findDeallocs() {
550  for (auto &entry : allocs) {
551  auto userIt =
552  llvm::find_if(entry.allocValue.getUsers(), [&](Operation *user) {
553  auto effectInterface = dyn_cast<MemoryEffectOpInterface>(user);
554  if (!effectInterface)
555  return false;
556  // Try to find a free effect that is applied to one of our values
557  // that will be automatically freed by our pass.
558  SmallVector<MemoryEffects::EffectInstance, 2> effects;
559  effectInterface.getEffectsOnValue(entry.allocValue, effects);
560  return llvm::any_of(
561  effects, [&](MemoryEffects::EffectInstance &it) {
562  return isa<MemoryEffects::Free>(it.getEffect());
563  });
564  });
565  // Assign the associated dealloc operation (if any).
566  if (userIt != entry.allocValue.user_end())
567  entry.deallocOperation = *userIt;
568  }
569  }

571  /// Finds correct dealloc positions according to the algorithm described at
572  /// the top of the file for all alloc nodes and block arguments that can be
573  /// handled by this analysis.
574  void placeDeallocs() const {
575  // Move or insert deallocs using the previously computed information.
576  // These deallocations will be linked to their associated allocation nodes
577  // since they don't have any aliases that can (potentially) increase their
578  // liveness.
579  for (const AllocEntry &entry : allocs) {
580  Value alloc = entry.allocValue;
581  auto aliasesSet = aliases.resolve(alloc);
582  assert(aliasesSet.size() > 0 && "must contain at least one alias");
583 
584  // Determine the actual block to place the dealloc and get liveness
585  // information.
586  Block *placementBlock =
587  findCommonDominator(alloc, aliasesSet, postDominators);
588  const LivenessBlockInfo *livenessInfo =
589  liveness.getLiveness(placementBlock);
590 
591  // We have to ensure that the dealloc will be after the last use of all
592  // aliases of the given value. We first assume that there are no uses in
593  // the placementBlock and that we can safely place the dealloc at the
594  // beginning.
595  Operation *endOperation = &placementBlock->front();
596  // Iterate over all aliases and ensure that the endOperation will point
597  // to the last operation of all potential aliases in the placementBlock.
598  for (Value alias : aliasesSet) {
599  Operation *aliasEndOperation =
600  livenessInfo->getEndOperation(alias, endOperation);
601  // Check whether the aliasEndOperation lies in the desired block and
602  // whether it is behind the current endOperation. If yes, this will be
603  // the new endOperation.
604  if (aliasEndOperation->getBlock() == placementBlock &&
605  endOperation->isBeforeInBlock(aliasEndOperation))
606  endOperation = aliasEndOperation;
607  }
608  // endOperation is the last operation behind which we can safely store
609  // the dealloc taking all potential aliases into account.
610 
611  // If there is an existing dealloc, move it to the right place.
612  if (entry.deallocOperation) {
613  entry.deallocOperation->moveAfter(endOperation);
614  } else {
615  // If the Dealloc position is at the terminator operation of the
616  // block, then the value should escape from a deallocation.
617  Operation *nextOp = endOperation->getNextNode();
618  if (!nextOp)
619  continue;
620  // If there is no dealloc node, insert one in the right place.
621  OpBuilder builder(nextOp);
622  builder.create<DeallocOp>(alloc.getLoc(), alloc);
623  }
624  }

627  /// Finds a common dominator for the given value while taking the positions
628  /// of the values in the value set into account. It supports dominator and
629  /// post-dominator analyses via template arguments.
630  template <typename DominatorT>
631  Block *findCommonDominator(Value value, const ValueSetT &values,
632  const DominatorT &doms) const {
633  // Start with the current block the value is defined in.
634  Block *dom = value.getParentBlock();
635  // Iterate over all aliases and their uses to find a safe placement block
636  // according to the given dominator information.
637  for (Value childValue : values)
638  for (Operation *user : childValue.getUsers()) {
639  // Move upwards in the dominator tree to find an appropriate
640  // dominator block that takes the current use into account.
641  dom = doms.findNearestCommonDominator(dom, user->getBlock());
642  }
643  return dom;
644  }

646  /// The operation this transformation was constructed from.
647  Operation *operation;
648 
649  /// Alias information that can be updated during the insertion of copies.
650  BufferPlacementAliasAnalysis aliases;
651 
652  /// Maps allocation nodes to their associated blocks.
653  AllocEntryList allocs;
654 
655  /// The underlying liveness analysis to compute fine grained information
656  /// about alloc and dealloc positions.
657  Liveness liveness;
658 
659  /// The dominator analysis to place deallocs in the appropriate blocks.
660  DominanceInfo dominators;
661 
662  /// The post dominator analysis to place deallocs in the appropriate blocks.
663  PostDominanceInfo postDominators;
664 };

754 /// Performs the actual function signature rewriting step.
755 LogicalResult BufferAssignmentFuncOpConverter::matchAndRewrite(
756  mlir::FuncOp funcOp, ArrayRef<Value> operands,
757  ConversionPatternRewriter &rewriter) const {
758  auto funcType = funcOp.getType();
759 
760  // Convert function arguments using the provided TypeConverter.
761  TypeConverter::SignatureConversion conversion(funcType.getNumInputs());
762  for (auto argType : llvm::enumerate(funcType.getInputs())) {
763  SmallVector<Type, 2> decomposedTypes, convertedTypes;
764  converter->tryDecomposeType(argType.value(), decomposedTypes);
765  converter->convertTypes(decomposedTypes, convertedTypes);
766  conversion.addInputs(argType.index(), convertedTypes);
767  }
768 
769  // Convert the result types of the function.
770  SmallVector<Type, 2> newResultTypes;
771  newResultTypes.reserve(funcOp.getNumResults());
772  for (Type resultType : funcType.getResults()) {
773  SmallVector<Type, 2> originTypes;
774  converter->tryDecomposeType(resultType, originTypes);
775  for (auto origin : originTypes) {
776  Type converted = converter->convertType(origin);
777  auto kind = converter->getResultConversionKind(origin, converted);
778  if (kind == BufferAssignmentTypeConverter::AppendToArgumentsList)
779  conversion.addInputs(converted);
780  else
781  // kind = BufferAssignmentTypeConverter::KeepAsFunctionResult
782  newResultTypes.push_back(converted);
783  }
784  }
785 
786  if (failed(rewriter.convertRegionTypes(&funcOp.getBody(), *converter,
787  &conversion)))
788  return failure();
789 
790  // Update the signature of the function.
791  rewriter.updateRootInPlace(funcOp, [&] {
792  funcOp.setType(rewriter.getFunctionType(conversion.getConvertedTypes(),
793  newResultTypes));
794  });
795  return success();
796 }

802 /// Performs the actual rewriting step.
803 LogicalResult BufferAssignmentCallOpConverter::matchAndRewrite(
804  CallOp callOp, ArrayRef<Value> operands,
805  ConversionPatternRewriter &rewriter) const {
806 
807  // This class represents a mapping from a result to a list of values and some
808  // results that have not yet constructed. Instead, the indices of these
809  // results in the operation that will be constructed are known. They will be
810  // replaced with the actual values when they are available. The order of
811  // adding to this mapping is important.
812  class ResultMapping {
813  public:
814  ResultMapping() { order = 0; };
815 
816  /// Add an available value to the mapping.
817  void addMapping(Value value) {
818  toValuesMapping.push_back({order++, value});
819  }
820 
821  /// Add the index of unavailble result value to the mapping.
822  void addMapping(unsigned index) {
823  toIndicesMapping.push_back({order++, index});
824  }
825 
826  /// This method returns the mapping values list. The unknown result values
827  /// that only their indicies are available are replaced with their values.
828  void getMappingValues(ValueRange valuesToReplaceIndices,
829  SmallVectorImpl<Value> &values) {
830  // Append available values to the list.
831  SmallVector<std::pair<unsigned, Value>, 2> res(toValuesMapping.begin(),
832  toValuesMapping.end());
833  // Replace the indices with the actual values.
834  llvm::for_each(
835  toIndicesMapping, [&](const std::pair<unsigned, unsigned> &entry) {
836  assert(entry.second < valuesToReplaceIndices.size() &&
837  "The value index is out of range.");
838  res.push_back({entry.first, valuesToReplaceIndices[entry.second]});
839  });
840  // Sort the values based on their adding orders.
841  llvm::sort(res, [](const std::pair<unsigned, Value> &v1,
842  const std::pair<unsigned, Value> &v2) {
843  return v1.first < v2.first;
844  });
845  // Fill the values.
846  llvm::for_each(res, [&](const std::pair<unsigned, Value> &entry) {
847  values.push_back(entry.second);
848  });
849  }
850 
851  private:
852  /// Keeping the inserting order of mapping values.
853  int order;
854 
855  /// Containing the mapping values with their inserting orders.
856  SmallVector<std::pair<unsigned, Value>, 2> toValuesMapping;
857 
858  /// Containing the indices of result values with their inserting orders.
859  SmallVector<std::pair<unsigned, unsigned>, 2> toIndicesMapping;
860  };
861 
862  Location loc = callOp.getLoc();
863  OpBuilder builder(callOp);
864  SmallVector<Value, 2> newOperands;
865 
866  // Create the operands list of the new `CallOp`. It unpacks the decomposable
867  // values if a decompose callback function has been provided by the user.
868  for (auto operand : operands) {
869  SmallVector<Value, 2> values;
870  this->converter->tryDecomposeValue(builder, loc, operand.getType(), operand,
871  values);
872  newOperands.append(values.begin(), values.end());
873  }
874 
875  // Create the new result types for the new `CallOp` and a mapping from the old
876  // result to new value(s).
877  SmallVector<Type, 2> newResultTypes;
878  SmallVector<ResultMapping, 4> mappings;
879  mappings.resize(callOp.getNumResults());
880  for (auto result : llvm::enumerate(callOp.getResults())) {
881  SmallVector<Type, 2> originTypes;
882  converter->tryDecomposeType(result.value().getType(), originTypes);
883  auto &resultMapping = mappings[result.index()];
884  for (Type origin : originTypes) {
885  Type converted = converter->convertType(origin);
886  auto kind = converter->getResultConversionKind(origin, converted);
887  if (kind == BufferAssignmentTypeConverter::KeepAsFunctionResult) {
888  newResultTypes.push_back(converted);
889  // The result value is not yet available. Its index is kept and it is
890  // replaced with the actual value of the new `CallOp` later.
891  resultMapping.addMapping(newResultTypes.size() - 1);
892  } else {
893  // kind = BufferAssignmentTypeConverter::AppendToArgumentsList
894  OpBuilder::InsertionGuard guard(rewriter);
895  rewriter.restoreInsertionPoint(
896  bufferAssignment->computeAllocPosition(result.value()));
897  MemRefType memref = converted.dyn_cast<MemRefType>();
898  if (!memref)
899  return callOp.emitError("Cannot allocate for a non-Memref type");
900  Value alloc = rewriter.create<AllocOp>(loc, memref);
901  newOperands.push_back(alloc);
902  resultMapping.addMapping(alloc);
903  }
904  }
905  }
906 
907  CallOp newCallOp = rewriter.create<CallOp>(loc, callOp.getCallee(),
908  newResultTypes, newOperands);
909 
910  // Build a replacing value for each result to replace its uses. If a result
911  // has multiple mapping values, it needs to be packed to a single value.
912  OpBuilder nextBuilder(callOp.getOperation()->getNextNode());
913  SmallVector<Value, 2> replacedValues;
914  replacedValues.reserve(callOp.getNumResults());
915  for (unsigned i = 0, e = callOp.getNumResults(); i < e; ++i) {
916  SmallVector<Value, 2> valuesToPack;
917  mappings[i].getMappingValues(newCallOp.getResults(), valuesToPack);
918  if (valuesToPack.empty()) {
919  // No replacement is required.
920  replacedValues.push_back(nullptr);
921  } else if (valuesToPack.size() == 1) {
922  replacedValues.push_back(valuesToPack.front());
923  } else {
924  // Values need to be packed using callback function. The same callback
925  // that is used for materializeArgumentConversion is used for packing.
926  Value packed = converter->materializeArgumentConversion(
927  nextBuilder, loc, callOp.getType(i), valuesToPack);
928  replacedValues.push_back(packed);
929  }
930  }
931  rewriter.replaceOp(callOp, replacedValues);
932  return success();
933 }
