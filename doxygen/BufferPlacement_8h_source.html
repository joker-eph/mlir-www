27 /// Prepares a buffer placement phase. It can place (user-defined) alloc
28 /// nodes. This simplifies the integration of the actual buffer-placement
29 /// pass. Sample usage:
30 /// BufferAssignmentPlacer baHelper(regionOp);
31 /// -> determine alloc positions
32 /// auto allocPosition = baHelper.computeAllocPosition(value);
33 /// -> place alloc
34 /// allocBuilder.setInsertionPoint(positions.getAllocPosition());
35 /// <create alloc>
36 /// Note: this class is intended to be used during legalization. In order
37 /// to move alloc and dealloc nodes into the right places you can use the
38 /// createBufferPlacementPass() function.
39 class BufferAssignmentPlacer {
40 public:
41  /// Creates a new assignment builder.
42  explicit BufferAssignmentPlacer(Operation *op);
43 
44  /// Returns the operation this analysis was constructed from.
45  Operation *getOperation() const { return operation; }
46 
47  /// Computes the actual position to place allocs for the given result.
48  OpBuilder::InsertPoint computeAllocPosition(OpResult result);
49 
50 private:
51  /// The operation this analysis was constructed from.
52  Operation *operation;
53 };

55 /// A helper type converter class for using inside Buffer Assignment operation
56 /// conversion patterns. The default constructor keeps all the types intact
57 /// except for the ranked-tensor types which is converted to memref types.
58 class BufferAssignmentTypeConverter : public TypeConverter {
59 public:
60  /// This enum is for showing how buffer placement operation converters should
61  /// conduct with certain result type after type conversion. This value can be
62  /// set/get for each specific type using setResultConversionKind or
63  /// getResultConversionKind.
64  enum ResultConversionKind { AppendToArgumentsList, KeepAsFunctionResult };
65 
66  BufferAssignmentTypeConverter();

68  /// This method tries to decompose a value of a certain type using provided
69  /// decompose callback functions. If it is unable to do so, the original value
70  /// is returned.
71  void tryDecomposeValue(OpBuilder &, Location, Type, Value,
72  SmallVectorImpl<Value> &);
73 
74  /// This method tries to decompose a type using provided decompose callback
75  /// functions. If it is unable to do so, the original type is returned.
76  void tryDecomposeType(Type, SmallVectorImpl<Type> &);
77 
78  /// This method registers a callback function that will be called to decompose
79  /// a value of a certain type into several values.
80  template <typename FnT,
81  typename T = typename llvm::function_traits<FnT>::template arg_t<2>>
82  void addDecomposeValueConversion(FnT &&callback) {
83  decomposeValueConversions.emplace_back(
84  wrapDecomposeValueConversionCallback<T>(std::forward<FnT>(callback)));
85  }
86 
87  /// This method registers a callback function that will be called to decompose
88  /// a type into several types.
89  template <typename FnT,
90  typename T = typename llvm::function_traits<FnT>::template arg_t<0>>
91  void addDecomposeTypeConversion(FnT &&callback) {
92  auto wrapper =
93  wrapDecomposeTypeConversionCallback<T>(std::forward<FnT>(callback));
94  decomposeTypeConversions.emplace_back(wrapper);
95  addConversion(std::forward<FnT>(callback));
96  }
97 
98  /// This method returns ResultConversionKind for the mapping from `origin`
99  /// type to `input` type.
100  ResultConversionKind getResultConversionKind(Type origin, Type input);
101 
102  /// This method registers ResultConversionKind for the mapping from type 'T'
103  /// to type 'U'.
104  template <typename T, typename U>
105  void setResultConversionKind(ResultConversionKind kind) {
106  assert((kind != AppendToArgumentsList ||
107  llvm::is_one_of<U, MemRefType, UnrankedMemRefType>::value) &&
108  "Only the memref typed values can be set to be appended to the "
109  "function argument list at the moment");
110  resultTypeConversions.emplace_back(
111  [=](Type origin, Type input) -> Optional<ResultConversionKind> {
112  if (origin.template isa<T>() && input.template isa<U>())
113  return kind;
114  return llvm::None;
115  });
116  }
117 
118 private:
119  using DecomposeValueConversionCallFn = std::function<Optional<LogicalResult>(
120  OpBuilder &, Location, Type, Value, SmallVectorImpl<Value> &)>;
121 
122  using DecomposeTypeConversionCallFn =
123  std::function<Optional<LogicalResult>(Type, SmallVectorImpl<Type> &)>;
124 
125  using ResultConversionKindFn =
126  std::function<Optional<ResultConversionKind>(Type, Type)>;
127 
128  /// Generate a wrapper for the given decompose value conversion callback.
129  template <typename T, typename FnT>
130  DecomposeValueConversionCallFn
131  wrapDecomposeValueConversionCallback(FnT &&callback) {
132  return [callback = std::forward<FnT>(callback)](
133  OpBuilder &builder, Location loc, Type type, Value value,
134  SmallVectorImpl<Value> &newValues) -> Optional<LogicalResult> {
135  if (T derivedType = type.dyn_cast<T>())
136  return callback(builder, loc, derivedType, value, newValues);
137  return llvm::None;
138  };
139  }
140 
141  /// Generate a wrapper for the given decompose type conversion callback.
142  template <typename T, typename FnT>
143  DecomposeTypeConversionCallFn
144  wrapDecomposeTypeConversionCallback(FnT &&callback) {
145  return [callback = std::forward<FnT>(callback)](
146  Type type,
147  SmallVectorImpl<Type> &results) -> Optional<LogicalResult> {
148  T derivedType = type.dyn_cast<T>();
149  if (!derivedType)
150  return llvm::None;
151  return callback(derivedType, results);
152  };
153  }
154 
155  SmallVector<ResultConversionKindFn, 2> resultTypeConversions;
156  SmallVector<DecomposeValueConversionCallFn, 2> decomposeValueConversions;
157  SmallVector<DecomposeTypeConversionCallFn, 2> decomposeTypeConversions;
158 };

203 /// Rewrites the `ReturnOp` to conform with the changed function signature.
204 /// Operands that correspond to return values and their types have been set to
205 /// AppendToArgumentsList are dropped. In their place, a corresponding copy
206 /// operation from the operand to the target function argument is inserted.
207 template <typename ReturnOpSourceTy, typename ReturnOpTargetTy,
208  typename CopyOpTy>
209 class BufferAssignmentReturnOpConverter
210  : public BufferAssignmentOpConversionPattern<ReturnOpSourceTy> {
211 public:
212  using BufferAssignmentOpConversionPattern<
213  ReturnOpSourceTy>::BufferAssignmentOpConversionPattern;
214 
215  /// Performs the actual return-op conversion step.
216  LogicalResult
217  matchAndRewrite(ReturnOpSourceTy returnOp, ArrayRef<Value> operands,
218  ConversionPatternRewriter &rewriter) const final {
219  Location loc = returnOp.getLoc();
220 
221  // Split the operands depending on whether they need a copy operation or
222  // they remain as operands of the return operation. If an operand is
223  // decomposable and a decompose callback function has been provided by the
224  // user, it will be unpacked.
225  SmallVector<Value, 2> newOperands, needCopyOperands;
226  OpBuilder builder(returnOp);
227  for (auto operand : llvm::enumerate(operands)) {
228  SmallVector<Value, 2> values;
229  this->converter->tryDecomposeValue(
230  builder, loc, operand.value().getType(), operand.value(), values);
231  Type type = returnOp.getOperand(operand.index()).getType();
232  SmallVector<Type, 2> originTypes;
233  this->converter->tryDecomposeType(type, originTypes);
234  for (auto value : llvm::enumerate(values)) {
235  Type origin = originTypes[value.index()];
236  Type converted = value.value().getType();
237  auto kind = this->converter->getResultConversionKind(origin, converted);
238  if (kind == BufferAssignmentTypeConverter::KeepAsFunctionResult)
239  newOperands.push_back(value.value());
240  else
241  // kind = BufferAssignmentTypeConverter::AppendToArgumentsList
242  needCopyOperands.push_back(value.value());
243  }
244  }
245 
246  // Insert Copy operations instead for the operands that have been removed
247  // from operand list and appended to the function arguments list.
248  Block &entryBlock = returnOp.getParentRegion()->front();
249  unsigned numFuncArgs = entryBlock.getNumArguments();
250  if (needCopyOperands.size() > numFuncArgs)
251  return returnOp.emitError(
252  "The number of operands that need Copy operations is more "
253  "than the number of target function arguments.");
254  unsigned destArgNum = numFuncArgs - needCopyOperands.size();
255  rewriter.setInsertionPoint(returnOp);
256  for (Value operand : needCopyOperands) {
257  rewriter.create<CopyOpTy>(loc, operand,
258  entryBlock.getArgument(destArgNum));
259  ++destArgNum;
260  }
261  rewriter.replaceOpWithNewOp<ReturnOpTargetTy>(returnOp, newOperands);
262  return success();
263  }
264 };
