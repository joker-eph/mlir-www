108 /// Accumulates into a python string from a method that accepts an
109 /// MlirStringCallback.
110 struct PyPrintAccumulator {
111  py::list parts;
112 
113  void *getUserData() { return this; }
114 
115  MlirStringCallback getCallback() {
116  return [](const char *part, intptr_t size, void *userData) {
117  PyPrintAccumulator *printAccum =
118  static_cast<PyPrintAccumulator *>(userData);
119  py::str pyPart(part, size); // Decodes as UTF-8 by default.
120  printAccum->parts.append(std::move(pyPart));
121  };
122  }
123 
124  py::str join() {
125  py::str delim("", 0);
126  return delim.attr("join")(parts);
127  }
128 };
129 
130 /// Accumulates into a python string from a method that is expected to make
131 /// one (no more, no less) call to the callback (asserts internally on
132 /// violation).
133 struct PySinglePartStringAccumulator {
134  void *getUserData() { return this; }
135 
136  MlirStringCallback getCallback() {
137  return [](const char *part, intptr_t size, void *userData) {
138  PySinglePartStringAccumulator *accum =
139  static_cast<PySinglePartStringAccumulator *>(userData);
140  assert(!accum->invoked &&
141  "PySinglePartStringAccumulator called back multiple times");
142  accum->invoked = true;
143  accum->value = py::str(part, size);
144  };
145  }
146 
147  py::str takeValue() {
148  assert(invoked && "PySinglePartStringAccumulator not called back");
149  return std::move(value);
150  }
151 
152 private:
153  py::str value;
154  bool invoked = false;
155 };

222 /// CRTP base classes for Python attributes that subclass Attribute and should
223 /// be castable from it (i.e. via something like StringAttr(attr)).
224 template <typename T>
225 class PyConcreteAttribute : public PyAttribute {
226 public:
227  // Derived classes must define statics for:
228  // IsAFunctionTy isaFunction
229  // const char *pyClassName
230  using ClassTy = py::class_<T, PyAttribute>;
231  using IsAFunctionTy = int (*)(MlirAttribute);
232 
233  PyConcreteAttribute() = default;
234  PyConcreteAttribute(MlirAttribute attr) : PyAttribute(attr) {}
235  PyConcreteAttribute(PyAttribute &orig)
236  : PyConcreteAttribute(castFrom(orig)) {}
237 
238  static MlirAttribute castFrom(PyAttribute &orig) {
239  if (!T::isaFunction(orig.attr)) {
240  auto origRepr = py::repr(py::cast(orig)).cast<std::string>();
241  throw SetPyError(PyExc_ValueError,
242  llvm::Twine("Cannot cast attribute to ") +
243  T::pyClassName + " (from " + origRepr + ")");
244  }
245  return orig.attr;
246  }
247 
248  static void bind(py::module &m) {
249  auto cls = ClassTy(m, T::pyClassName);
250  cls.def(py::init<PyAttribute &>(), py::keep_alive<0, 1>());
251  T::bindDerived(cls);
252  }
253 
254  /// Implemented by derived classes to add methods to the Python subclass.
255  static void bindDerived(ClassTy &m) {}
256 };

588 void mlir::python::populateIRSubmodule(py::module &m) {
589  // Mapping of MlirContext
590  py::class_<PyMlirContext>(m, "Context")
591  .def(py::init<>())
592  .def(
593  "parse_module",
594  [](PyMlirContext &self, const std::string module) {
595  auto moduleRef =
596  mlirModuleCreateParse(self.context, module.c_str());
597  // TODO: Rework error reporting once diagnostic engine is exposed
598  // in C API.
599  if (mlirModuleIsNull(moduleRef)) {
600  throw SetPyError(
601  PyExc_ValueError,
602  "Unable to parse module assembly (see diagnostics)");
603  }
604  return PyModule(moduleRef);
605  },
606  py::keep_alive<0, 1>(), kContextParseDocstring)
607  .def(
608  "parse_attr",
609  [](PyMlirContext &self, std::string attrSpec) {
610  MlirAttribute type =
611  mlirAttributeParseGet(self.context, attrSpec.c_str());
612  // TODO: Rework error reporting once diagnostic engine is exposed
613  // in C API.
614  if (mlirAttributeIsNull(type)) {
615  throw SetPyError(PyExc_ValueError,
616  llvm::Twine("Unable to parse attribute: '") +
617  attrSpec + "'");
618  }
619  return PyAttribute(type);
620  },
621  py::keep_alive<0, 1>())
622  .def(
623  "parse_type",
624  [](PyMlirContext &self, std::string typeSpec) {
625  MlirType type = mlirTypeParseGet(self.context, typeSpec.c_str());
626  // TODO: Rework error reporting once diagnostic engine is exposed
627  // in C API.
628  if (mlirTypeIsNull(type)) {
629  throw SetPyError(PyExc_ValueError,
630  llvm::Twine("Unable to parse type: '") +
631  typeSpec + "'");
632  }
633  return PyType(type);
634  },
635  py::keep_alive<0, 1>(), kContextParseTypeDocstring)
636  .def(
637  "get_unknown_location",
638  [](PyMlirContext &self) {
639  return PyLocation(mlirLocationUnknownGet(self.context));
640  },
641  py::keep_alive<0, 1>(), kContextGetUnknownLocationDocstring)
642  .def(
643  "get_file_location",
644  [](PyMlirContext &self, std::string filename, int line, int col) {
645  return PyLocation(mlirLocationFileLineColGet(
646  self.context, filename.c_str(), line, col));
647  },
648  py::keep_alive<0, 1>(), kContextGetFileLocationDocstring,
649  py::arg("filename"), py::arg("line"), py::arg("col"))
650  .def(
651  "create_region",
652  [](PyMlirContext &self) {
653  // The creating context is explicitly captured on regions to
654  // facilitate illegal assemblies of objects from multiple contexts
655  // that would invalidate the memory model.
656  return PyRegion(self.context, mlirRegionCreate(),
657  /*detached=*/true);
658  },
659  py::keep_alive<0, 1>(), kContextCreateRegionDocstring)
660  .def(
661  "create_block",
662  [](PyMlirContext &self, std::vector<PyType> pyTypes) {
663  // In order for the keep_alive extend the proper lifetime, all
664  // types must be from the same context.
665  for (auto pyType : pyTypes) {
666  if (!mlirContextEqual(mlirTypeGetContext(pyType.type),
667  self.context)) {
668  throw SetPyError(
669  PyExc_ValueError,
670  "All types used to construct a block must be from "
671  "the same context as the block");
672  }
673  }
674  llvm::SmallVector<MlirType, 4> types(pyTypes.begin(),
675  pyTypes.end());
676  return PyBlock(self.context,
677  mlirBlockCreate(types.size(), &types[0]),
